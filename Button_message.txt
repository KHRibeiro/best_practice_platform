  async function getSiteRegionByEmail(email) {
    const { data, error } = await supabase
      .from("users")
      .select("site_region")
      .eq("email", email)
      .single(); // retorna só um resultado

    if (error) {

    } else {
      console.log(`Site region de ${email}:`, data.site_region);

  }
